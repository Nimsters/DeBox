\newcommand{\rf}{{\bf reference}}
{\ttfamily
\noindent
PROOF $~~~~~$\tra {\bf title}: SEQUENT STEPS CONCLUSION\\
SEQUENT $~~~$\tra PREMISES GOAL\\
PREMISES $~~$\tra \txt{Without premises,} | \txt{From the premise }
        FORMULA\txt{,}
        | \txt{From the premises} PRMLIST\txt{,}\\
PRMLIST $~~~$\tra FORMULA\txt{, and } FORMULA | FORMULA\txt{, } PRMLIST\\
GOAL $~~~~~~$\tra \txt{we wish to prove } FORMULA\txt{.}\\
STEPS $~~~~~$\tra STEP | STEP STEPS\\
STEP $~~~~~~$\tra LINE [\rf]. | BOX\\
BOX $~~~~~~~$\tra OPEN CLOSE | OPEN STEPS CLOSE\\
OPEN $~~~~~~$\tra \txt{We now assume } FORMULA [\rf].\\
CLOSE $~~~~~$\tra ARGUMENT \txt{ we get the part-conclusion } FORMULA [\rf] 
                 \txt{from the }\\
$~~~~~~~~~~~~~~~$\txt{assumption } [\rf].\\
LINE $~~~~~~$\tra ARGUMENT \txt{ we get } FORMULA\\
CONCLUSION \tra ARGUMENT \txt{ we may conclude } FORMULA.\\
ARGUMENT $~~$\tra \txt{By applying } PATTERN\\
PATTERN $~~~$\tra LEM | RULE1\txt{ to }[\rf] 
            | RULE2\txt{ to }[\rf]\txt{ and }[\rf] |\\$~~~~~~~~~~~~~~$
        ORELM\txt{ to }[\rf]\txt{, }[\rf]-[\rf]\txt{, and }[\rf]-[\rf] |\\
$~~~~~~~~~~~~~~$ RULEB\txt{ to }[\rf]-[\rf]\\
RULE1 $~~~~~$\tra\txt{the first and-elimination rule} |
                 \txt{the second and-elimination rule} |\\
$~~~~~~~~~~~~~~~$\txt{the first or-introduction rule} |
                 \txt{the second or-introduction rule} |\\
$~~~~~~~~~~~~~~~$\txt{elimination of double-negation} |
                 \txt{introduction of double-negation} |\\
$~~~~~~~~~~~~~~~$\txt{elimination of absurdity}\\
RULE2 $~~~~~$\tra\txt{Modus Tollens} | \txt{the and-introduction rule} |
                 \txt{the implication-elimination rule} |\\
$~~~~~~~~~~~~~~~$\txt{the negation-elimination rule} |\\
RULEB $~~~~~$\tra\txt{the implication-introduction rule} |
                 \txt{the negation-introduction rule} |\\
$~~~~~~~~~~~~~~~$\txt{proof by contradiction}\\
ORELM $~~~~~$\tra\txt{the or-elimination rule}\\
LEM $~~~~~~~$\tra \txt{the law of the excluded middle}
}
