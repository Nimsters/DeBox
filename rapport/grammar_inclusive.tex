\newcommand{\id}{{\bf id}}
{\ttfamily
\noindent
PROOF $~~~~~$\tra \id: SEQUENT STEPS\\
SEQUENT $~~~$\tra \txt{We wish to prove} FORMULA. | 
                  \txt{From} PREMISES\txt{, we wish to prove} FORMULA.\\
PREMISES $~~$\tra \txt{the premise } FORMULA
                | \txt{the premises} PRMLIST\\
PRMLIST $~~~$\tra FORMULA \txt{and} FORMULA
                | FORMULA\txt{, } FORMULA\txt{, and} FORMULA 
                | FORMULA\txt{, } PRMLIST\\
STEPS $~~~~~$\tra STEP | STEP STEPS\\
STEP $~~~~~~$\tra RULE REFERENCES. | RULE CONCLUSION 
                | RULE \txt{to} REFERENCES CONCLUSION |\\
$~~~~~~~~~~~~~~$  RULE REFERENCES CONCLUSION\\
CONCLUSION   \tra \txt{, we get} FORMULA [\id]. 
                | \txt{, we conclude} FORMULA.\\
REFERENCES   \tra REF | REFLIST\\
REFLIST $~~~$\tra REF\txt{, }REF\txt{, and} REF | REF\txt{ and }REF |
                  REF\txt{, }REFLIST\\
REF $~~~~~~~$\tra [\id] | [\id]\txt{ - }[\id]\\
RULE $~~~~~~$\tra CPY | AIN | AE1 | AE2 | OI1 | OI2 | OEL | IIN | IEL |\\
$~~~~~~~~~~~~~~$  NIN | NEL | DIN | DEL | BEL | MOD | PBC | LEM\\
PRM $~~~~~~~$\tra\txt{We have the premise}\\
ASS $~~~~~~~$\tra\txt{Assume}\\
DIS $~~~~~~~$\tra\txt{Discharge assumption}\\
CPY $~~~~~~~$\tra\txt{By copying}\\
AIN $~~~~~~~$\tra\txt{By applying the and-introduction rule}\\
AE1 $~~~~~~~$\tra\txt{By applying the first and-elimination rule}\\
AE2 $~~~~~~~$\tra\txt{By applying the second and-elimination rule}\\
OI1 $~~~~~~~$\tra\txt{By applying the first or-introduction rule}\\
OI2 $~~~~~~~$\tra\txt{By applying the second or-introduction rule}\\
OEL $~~~~~~~$\tra\txt{By applying the or-elimination rule}\\
IIN $~~~~~~~$\tra\txt{By applying the implication-introduction rule}\\
IEL $~~~~~~~$\tra\txt{By applying the implication-elimination rule}\\
NIN $~~~~~~~$\tra\txt{By applying the negation-introduction rule}\\
NEL $~~~~~~~$\tra\txt{By applying the negation-elimination rule}\\
DIN $~~~~~~~$\tra\txt{By applying introduction of double-negation}\\
DEL $~~~~~~~$\tra\txt{By applying elimination of double-negation}\\
BEL $~~~~~~~$\tra\txt{By applying elimination of absurdity}\\
MOD $~~~~~~~$\tra\txt{By applying Modus Tollens}\\
PBC $~~~~~~~$\tra\txt{By applying proof by contradiction}\\
LEM $~~~~~~~$\tra\txt{By applying the law of the excluded middle}\\
}
