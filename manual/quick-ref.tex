\paragraph{Sequents} ~
\begin{itemize}\itemsep0pt
\item We wish to prove \f{<formula>}.
\item From the premise \f{<premise>}, we wish to prove \f{<formula>}.
\item From the premises \f{<premise 1, ...,>} and 
        \f{<premise n>}, we wish to prove \f{<formula>}.
\end{itemize}
